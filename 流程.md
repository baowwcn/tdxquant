graph TD

    %% 0. 模块导入

    end

    %% 2. 全局变量
    subgraph GLOBALS [🌍 全局状态]
        direction TB
        class GLOBALS global
        G1[SUBSCRIBE_CODES: []] --> G2[last_warn_time: defaultdict]
        G2 --> G3[EXIT_FLAG: False]
        G3 --> G4[TRIGGERED_STOCKS: set()]
    end

    %% 3. 信号处理器
    subgraph HANDLER [🛑 信号处理器]
        direction TB
        class HANDLER handler
        SIG[signal_handler] -->|捕获 Ctrl+C| S1[设置 EXIT_FLAG=True]

    %% 4. 核心函数组
    subgraph FUNCTIONS [🔧 核心函数]
        direction LR

        subgraph GET_CODES [获取股票代码]
            direction TB
            class GET_CODES func
            GC1[遍历板块] --> GC2{股票有效?}
            GC2 -->|是| GC3[加入 valid_codes]
            GC2 -->|否| GC4[跳过]
            GC3 --> GC5[返回代码列表]
        end

        subgraph SUBSCRIBE [批量订阅]
            direction TB
            class SUBSCRIBE func
            SUB1[分批: BATCH_SIZE] --> SUB2[subscribe_hq]
            SUB2 --> SUB3[回调: on_data]
        end

        subgraph UNSUB [取消订阅]
            direction TB
            class UNSUB func
            US1[单个取消] --> US2[全部取消]
        end

    subgraph ONDATA [⚡ on_data 回调 - 核心策略]

        OD3 -->|无| OD3A[return None]
        OD3 -->|有| OD4{已触发过?}

        OD4 -->|是| OD4A[return None]

        OD5 -->|是| OD5A[return None]

        OD7 -->|否| OD7A[return None]
        OD7 -->|是| OD8[计算涨幅%]

        D1 -->|否| END1[结束处理]

            A0[记录时间] --> A1[加入TRIGGERED集合]

            A2 --> A3[构造预警信息]
            A3 --> A4[发送通达信预警]
            A4 --> A5[发送钉钉通知]
        end
    end

    %% 6. 主程序流程
    subgraph MAIN [🎬 主程序入口]
        direction TB
        class MAIN main

        M0[注册信号处理器] --> M1[TDX初始化]

        M1 -->|成功| M2[钉钉通知启动]
        M1 -->|失败| M2A[退出程序]

        M2 --> M3[获取股票代码]
        M3 -->|空| M3A[退出程序]
        M3 -->|有数据| M4[批量订阅]

        M4 --> M5[打印配置信息]
        M5 --> M6[主循环]

            L2 --> L3{收到信号?}
            L3 -->|是| L4[退出循环]
            L3 -->|否| L2
        end

        L4 --> M7[取消所有订阅]
        M7 --> M8[打印退出消息]
    end

    %% 拓扑连接

    HANDLER --> FUNCTIONS

### 🔑 关键路径说明

| 路径 | 说明 |
|------|------|
| **绿线** 🟢 | 初始化流程（单向） |
| **蓝线** 🔵 | 订阅/取消订阅机制 |
| **橙线** 🟠 | **核心策略决策**（涨幅计算 → 阈值判断 → 预警触发） |
| **红线** 🔴 | 退出清理流程 |

### ⚡ 数据流向

```
TDX推送 → on_data回调 → JSON解析 → 过滤检查 → 涨幅计算 
    ↓
[>5%?] --否--> 结束
    ↓是
触发预警 → 发送信讯 → 钉钉通知 → 取消订阅
```

### 🎯 核心决策点

1. **股票有效性过滤** (.SH/.SZ后缀)
2. **防抖时间窗** (10秒)
3. **涨幅阈值** (5%)
4. **单次触发** (TRIGGERED_STOCKS 集合)
5. **自动退订** (触发后立即取消)
